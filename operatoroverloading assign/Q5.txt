//5) Define a class with constructor and  member function
//
//	create object, call member function
//	now create a reference to that object, and using that reference try to call member function
//
	
#include<iostream>
using namespace std;

class myclass
{
private:
	int k;
public:
	myclass(int k)
	{
		this->k=k;
	}
	void disp1()
	{
     cout<<"in disp1"<<k<<endl;
	}
	
};


void main()
{
	myclass m1(5);
	m1.disp1();
	myclass &ref=m1;
	ref.disp1();
}
